070        static abstract class CLEntity<T extends PointerType> {
071    
072            private T entity;
073    
074            public CLEntity(T entity) {
075                if (entity == null) {
076                    throw new IllegalArgumentException("Null OpenCL " + getClass().getSimpleName() + " !");
077                }
078                this.entity = entity;
079            }
080    
081            public T get() {
082                return entity;
083            }
084    
085            public Pointer getPointer() {
086                return entity.getPointer();
087            }
088    
089            @Override
090            protected void finalize() throws Throwable {
091                clear();
092                entity = null;
093            }
094    
095            protected abstract void clear();
096        }

214            public CLProgram build() throws CLBuildException {
215    
216                int err = CL.clBuildProgram(get(), 0, null/*context.getDeviceIds()*/, (String) null, null, null);
217                if (err != CL_SUCCESS) {
218                    NativeLongByReference len = new NativeLongByReference();
219                    int bufLen = 2048;
220                    Memory buffer = new Memory(bufLen);
221    
222                    HashSet<String> errs = new HashSet<String>();
223                    for (cl_device_id device_id : context.getDeviceIds()) {
224                        error(CL.clGetProgramBuildInfo(get(), device_id, CL_PROGRAM_BUILD_LOG, toNL(bufLen), buffer, len));
225                        String s = buffer.getString(0);
226                        errs.add(s);
227                    }
228                    throw new CLBuildException(errorString(err), errs);
229                }
230                return this;
231            }
232    
233            @Override
234            protected void clear() {
235                CL.clReleaseProgram(get());
236            }

267    
268                if (arg instanceof NativeLong) {

270                } else if (arg instanceof CLMem) {

272                } else if (arg instanceof Integer) {
273                    setArg(i, (Integer) arg);
274                } else if (arg instanceof Long) {

276                } else if (arg instanceof Short) {

278                } else if (arg instanceof Byte) {

282                } else if (arg instanceof Double) {
283                    setArg(i, (Double) arg);
284                } else {
285                    throw new IllegalArgumentException("Cannot handle kernel arguments of type " + arg.getClass().getName() + ". Use CLKernel.get() and OpenCL4Java directly.");

329            /// TODO: Get the maximum work-group size with CL.clGetKernelWorkGroupInfo(CL_KERNEL_WORK_GROUP_SIZE)
330            public void enqueueNDRange(CLQueue queue, int[] globalSizes, int[] localSizes) {
331                int nDims = globalSizes.length;
332                if (localSizes.length != nDims) {
333                    throw new IllegalArgumentException("Global and local sizes must have same dimensions, given " + globalSizes.length + " vs. " + localSizes.length);
334                }
335                NativeLong[] globalSizesNL = new NativeLong[nDims], localSizesNL = new NativeLong[nDims];
336                for (int i = 0; i < nDims; i++) {
337                    globalSizesNL[i] = toNL(globalSizes[i]);
338                    localSizesNL[i] = toNL(localSizes[i]);
339                }
340                error(CL.clEnqueueNDRangeKernel(queue.get(), get(), 1, null, globalSizesNL, localSizesNL, 0, null, null));
341            }

369            public CLProgram createProgram(String... srcs) {
370    
371                String[] source = new String[srcs.length];
372                NativeLong[] lengths = new NativeLong[srcs.length];
373                for (int i = 0; i < srcs.length; i++) {
374                    source[i] = srcs[i];
375                    lengths[i] = toNL(srcs[i].length());
376                }
377                IntBuffer errBuff = IntBuffer.wrap(new int[1]);
378                cl_program program = CL.clCreateProgramWithSource(get(), srcs.length, source, lengths, errBuff);
379                error(errBuff.get(0));
380                return new CLProgram(this, program);
381            }

426            @SuppressWarnings("deprecation")
427            protected CLMem createMem(final Buffer buffer, int byteCount, final int clMemFlags, final boolean retainBufferReference) {
428                if (buffer != null) {
429                    byteCount = getByteCount(buffer);
430                } else if (retainBufferReference) {
431                    throw new IllegalArgumentException("Cannot retain reference to null pointer !");
432                }
433    
434                if (byteCount <= 0) {
435                    throw new IllegalArgumentException("Buffer size must be greater than zero (asked for size " + byteCount + ")");
436                }
437    
438                IntByReference errRef = new IntByReference();
439                //IntBuffer errBuff = IntBuffer.wrap(new int[1]);
440                cl_mem mem = CL.clCreateBuffer(
441                        get(),
442                        clMemFlags,
443                        toNL(byteCount),
444                        buffer == null ? null : Native.getDirectBufferPointer(buffer),
445                        errRef);
446                error(errRef.getValue());
447    
448                return new CLMem(this, byteCount, mem) {
449                    /// keep a hard reference to the buffer
450    
451                    public Buffer b = retainBufferReference ? buffer : null;
452    
453                    @Override
454                    public String toString() {
455                        return "CLMem(flags = " + clMemFlags + (b == null ? "" : ", " + getByteCount(b) + " bytes") + ")";
456                    }
457                };
458            }

521        public static int getByteCount(Buffer b) {
522            int c = b.capacity();
523            if (b instanceof IntBuffer || b instanceof FloatBuffer) {
524                return c * 4;
525            }
526            if (b instanceof LongBuffer || b instanceof DoubleBuffer) {
527                return c * 8;
528            }
529            if (b instanceof ShortBuffer || b instanceof CharBuffer) {
530                return c * 2;
531            }
532            if (b instanceof ByteBuffer) {
533                return c;
534            }
535            throw new UnsupportedOperationException("Cannot guess byte size of buffers of type " + b.getClass().getName());
536        }

548        static String errorString(int err) {
549            if (err == CL_SUCCESS) {
550                return null;
551            }
552            List<String> candidates = new ArrayList<String>();
553            for (Field f : OpenCLLibrary.class.getDeclaredFields()) {
554                if (!Modifier.isStatic(f.getModifiers())) {
555                    continue;
556                }
557                if (f.getType().equals(Integer.TYPE)) {
558                    try {
559                        int i = (Integer) f.get(null);
560                        if (i == err) {
561                            candidates.add(f.getName());
562                        }
563                    } catch (Exception e) {
564                        e.printStackTrace();
565                    }
566                }
567            }
568            return StringUtils.implode(candidates, " or ");
569        }

571        static void error(int err) {
572            String str = errorString(err);
573            if (str == null) {
574                return;
575            }
576            throw new RuntimeException("OpenCL Error : " + str + " (code " + err + ")");
577        }
