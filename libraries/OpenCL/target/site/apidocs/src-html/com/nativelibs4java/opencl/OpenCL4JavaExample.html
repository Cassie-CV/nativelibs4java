001    package com.nativelibs4java.opencl;
002    
003    import java.nio.*;
004    import static com.nativelibs4java.opencl.OpenCL4Java.*;
005    
006    /// @see http://ati.amd.com/technology/streamcomputing/intro_opencl.html#simple
007    public class OpenCL4JavaExample {
008    
009        public static void main(String[] args) {
010            try {
011                int dataSize = 128;
012    
013                FloatBuffer data = floatBuffer(dataSize);
014                FloatBuffer resultsf = floatBuffer(dataSize);
015                IntBuffer resultsi = intBuffer(dataSize);
016    
017                for (int i = 0; i < dataSize; i++) {
018                    data.put(i, i);
019                }
020    
021                CLDevice[] devices = CLDevice.listAllDevices();
022                for (CLDevice device : devices) {
023                    System.out.println("[OpenCL] Found device \"" + device.getName() + "\"");
024                    System.out.println(device.getExecutionCapabilities());
025                }
026                CLContext context = CLContext.createContext(devices);
027    
028                String src = "\n" +
029                        "__kernel square(                                                       \n" +
030                        "   __global const float* input,                                              \n" +
031                        "   __global float* outputf,                                             \n" +
032                        "   __global int* outputi,                                             \n" +
033                        "   const unsigned int count)                                           \n" +
034                        "{                                                                      \n" +
035                        "   int i = get_global_id(0);                                           \n" +
036                        "   if(i < count)                                                       \n" +
037                        "       outputf[i] = input[i] * input[i];                                \n" +
038                        "       outputi[i] = i;// + input[i] * input[i];                                \n" +
039                        "}                                                                      \n" +
040                        "\n";
041    
042                CLProgram program = context.createProgram(src).build();
043                CLKernel kernel = program.createKernel(
044                        "square",
045                        context.createInput(data, false),
046                        context.createOutput(resultsf),
047                        context.createOutput(resultsi),
048                        dataSize);
049    
050                CLQueue queue = context.createDefaultQueue();
051                kernel.enqueueNDRange(queue, new int[]{dataSize}, new int[]{1});
052                queue.finish();
053    
054                for (int iRes = 0; iRes < dataSize; iRes++) {
055                    float d = data.get(iRes), rf = resultsf.get(iRes);
056                    int ri = resultsi.get(iRes);
057                    System.out.println(d + "\t->\tfloat: " + rf + ", int: " + ri);
058                }
059            } catch (CLBuildException e) {
060                e.printStackTrace();
061            }
062        }
063    }
